async function voerControleUit() { const rawInput = document.getElementById('coords').value; const radiusKm = parseFloat(document.getElementById('radiusInput').value) || 6; const statusBox = document.getElementById('statusBox'); const tijd = new Date().toLocaleTimeString(); statusBox.className = "status"; statusBox.innerText = `[${tijd}] Invoer verwerken...`; const parsed = await parseInputNaarCoordinaten(rawInput); if (!parsed) { alert("Kon de locatie niet herkennen! Gebruik de GPS-knop of voer coördinaten in."); stopCheck(); return; } const myLat = parsed.lat; const myLon = parsed.lon; statusBox.innerText = `[${tijd}] Ophalen data...\nLocatie: ${myLat.toFixed(4)}, ${myLon.toFixed(4)}\nStraal: ${radiusKm} km`; try { const response = await fetch(`${WORKER_URL}?lat=${myLat}&lon=${myLon}&radius=${radiusKm}`); const data = await response.json(); console.log("Ontvangen data van EuRIS:", data); // Bepaal de lijst met schepen uit de respons let schepen = []; if (Array.isArray(data)) schepen = data; else if (Array.isArray(data.vessels)) schepen = data.vessels; else if (Array.isArray(data.items)) schepen = data.items; else if (Array.isArray(data.positions)) schepen = data.positions; else if (Array.isArray(data.data)) schepen = data.data; let bootGevonden = false; let gevondenBoodschap = ""; let aantalGevonden = 0; for (let ship of schepen) { // Haal de coördinaten op uit verschillende mogelijke EuRIS formats let shipLat = ship.latitude ?? ship.lat ?? (ship.position && ship.position.latitude) ?? (ship.geometry && ship.geometry.coordinates && ship.geometry.coordinates[1]); let shipLon = ship.longitude ?? ship.lon ?? (ship.position && ship.position.longitude) ?? (ship.geometry && ship.geometry.coordinates && ship.geometry.coordinates[0]); let snelheid = ship.speed ?? ship.sog ?? ship.speedOverGround ?? 0; let naam = ship.name || ship.vesselName || ship.shipName || `Schip (${ship.mmsi || ship.eni || 'Onbekend'})`; if (shipLat !== undefined && shipLon !== undefined) { const afstand = berekenAfstandKm(myLat, myLon, parseFloat(shipLat), parseFloat(shipLon)); // Als het schip binnen de straal valt (we negeren de snelheidsfilter tijdelijk om zeker te zijn dat alles gedetecteerd wordt) if (afstand <= radiusKm) { bootGevonden = true; aantalGevonden++; gevondenBoodschap += `🚨 ${naam}\nAfstand: ${afstand.toFixed(2)} km | Snelheid: ${snelheid} km/h\n\n`; } } } if (bootGevonden) { statusBox.className = "status alert-active"; statusBox.innerText = `[${tijd}] ALARM! (${aantalGevonden} schip/schepen nabij):\n\n${gevondenBoodschap}`; speelAlarmGeluid(); } else { statusBox.innerText = `[${tijd}] Laatste check OK.\nGeen schepen gedetecteerd binnen ${radiusKm} km.`; } } catch (error) { console.error(error); statusBox.innerText = `[${tijd}] Fout bij ophalen data:\n${error.message}`; } }
